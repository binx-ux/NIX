Load Script

--// Binxix Hub V6 Loader

local url = "https://raw.githubusercontent.com/binx-ux/airhub-binxix-v6/refs/heads/main/script/aimbot"

local success, err = pcall(function()
    loadstring(game:HttpGet(url, true))()
end)

if not success then
    warn("[BinxixHub] Failed to load script.")
    warn(err)
else
    print("[BinxixHub] Loaded successfully.")
end
